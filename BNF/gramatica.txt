<programa> ::= <comentario> <programa> 
             | <instrucao> $OPE_FIN$ <programa> 
             | <instrucao> $OPE_FIN$

<comentario> ::= $INST_HASH$ <texto_comentario>

<texto_comentario> ::= <letra_digito> 
                     | <letra_digito> <texto_comentario>

<instrucao> ::= <atribuicao> 
              | <declaracao> 
              | <condicional> 
              | <laco> 
              | <impressao>

<atribuicao> ::= <identificador> $OPE_ATR$ <expressao>

<declaracao> ::= <tipo> <identificador_lista>

<tipo> ::= $TIPO_INT$ | $TIPO_FLOAT$

<identificador_lista> ::= <identificador> 
                        | <identificador> $OPE_VIR$ <identificador_lista>

<condicional> ::= $INST_IF$ $ABR_PAR$ <condicao> $FEC_PAR$ $ABR_CHA$ <programa> $FEC_CHA$ 
                | $INST_IF$ $ABR_PAR$ <condicao> $FEC_PAR$ $ABR_CHA$ <programa> $FEC_CHA$ $INST_ELSE$ $ABR_CHA$ <programa> $FEC_CHA$

<laco> ::= $INST_WHILE$ $ABR_PAR$ <condicao> $FEC_PAR$ $ABR_CHA$ <programa> $FEC_CHA$

<impressao> ::= $INST_PRINT$ $ABR_PAR$ <identificador> $FEC_PAR$ 
              | $INST_PRINT$ <identificador>

<expressao> ::= <termo> 
              | <termo> <operador> <expressao>

<condicao> ::= <expressao> <operador_relacional> <expressao>

<termo> ::= <identificador> | <numero>

<identificador> ::= <letra> <identificador_aux>

<identificador_aux> ::= <letra_digito> 
                      | <letra_digito> <identificador_aux>

<numero> ::= <digito> 
           | <digito> $OPE_PTO$ <digito>

<letra_digito> ::= <letra> | <digito>

<letra> ::= $LETRA_MIN$ | $LETRA_MAI$

<digito> ::= $DIGITO$

<operador> ::= $OPE_ARI_SOM$ | $OPE_ARI_SUB$ | $OPE_ARI_MUL$ | $OPE_ARI_DIV$

<operador_relacional> ::= $OPE_REL_MAI$ | $OPE_REL_MEN$ | $OPE_REL_IGU$ | $OPE_REL_DIF$ | $OPE_REL_MAIG$ | $OPE_REL_MEIG$
